void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: MyToggleImageApp(),
    );
  }
}

class MyToggleImageApp extends StatefulWidget {
  @override
  _MyToggleImageAppState createState() => _MyToggleImageAppState();
}

class _MyToggleImageAppState extends State<MyToggleImageApp> {
  bool _isImage1 = true; // Initial image state
  int _counter = 0; // Counter variable from the previous example

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Toggle Image App'),
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            _isImage1
                ? Image.asset(
                    'assets/image1.png', // Replace with your image asset path
                    width: 150,
                    height: 150,
                  )
                : Image.asset(
                    'assets/image2.png', // Replace with your second image asset path
                    width: 150,
                    height: 150,
                  ),
            SizedBox(height: 20),
            ElevatedButton(
              onPressed: () {
                // Toggle between the two images when the button is pressed
                setState(() {
                  _isImage1 = !_isImage1;
                  _counter++; // Incrementing the counter as in the previous example
                });
              },
              child: Text('Toggle Image'),
            ),
            SizedBox(height: 10),
            Text(
              'Counter Value: $_counter',
              style: TextStyle(fontSize: 16),
            ),
          ],
        ),
      ),
    );
  }
}
